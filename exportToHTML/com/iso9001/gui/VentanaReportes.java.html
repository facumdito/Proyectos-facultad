124      private void generarReporte(ActionEvent e) {
125          String tipoReporte = (String) comboTipoReporte.getSelectedItem();
126          StringBuilder reporte = new StringBuilder();
127  
128          try {
129              switch (tipoReporte) {
130                  case "Estado General del Sistema":
131                      reporte.append(generarReporteEstadoGeneral());
132                      break;
133  
134                  case "Reporte de Procesos":
135                      reporte.append(generarReporteProcesos());
136                      break;
137  
138                  case "Reporte de Indicadores":
139                      reporte.append(generarReporteIndicadores());
140                      break;
141  
142                  case "Reporte de No Conformidades":
143                      reporte.append(generarReporteNoConformidades());
144                      break;
145  
146                  case "Reporte de Preparación para Auditoría":
147                      reporte.append(generarReporteAuditoria());
148                      break;
149  
150                  case "Análisis de Cumplimiento ISO 9001":
151                      reporte.append(generarAnalisisCumplimiento());
152                      break;
153  
154                  case "Dashboard Ejecutivo":
155                      reporte.append(generarDashboardEjecutivo());
156                      break;
157  
158                  default:
159                      reporte.append("Tipo de reporte no reconocido.");
160              }
161  
162              ultimoReporte = reporte.toString();
163              areaReporte.setText(ultimoReporte);
164              areaReporte.setCaretPosition(0);
165              btnExportar.setEnabled(true);
166  
167          } catch (Exception ex) {
168              JOptionPane.showMessageDialog(this,
169                      "Error al generar el reporte: " + ex.getMessage(),
170                      "Error", JOptionPane.ERROR_MESSAGE);
171          }
172      }

175          List<Proceso> procesos = procesoManager.obtenerTodosProcesos();
176          List<Indicador> indicadores = indicadorManager.obtenerTodosIndicadores();

182      private String generarReporteProcesos() {
183          StringBuilder reporte = new StringBuilder();
184  
185          reporte.append("REPORTE DETALLADO DE PROCESOS\n");
186          reporte.append("=".repeat(50)).append("\n");
187          reporte.append("Fecha: ").append(LocalDate.now()).append("\n\n");
188  
189          // Estadísticas generales
190          reporte.append("RESUMEN EJECUTIVO\n");
191          reporte.append("-".repeat(25)).append("\n");
192          reporte.append("Total de procesos: ").append(procesoManager.getTotalProcesos()).append("\n");
193          reporte.append("Procesos activos: ").append(procesoManager.getProcesosActivos()).append("\n");
194          reporte.append("Eficiencia promedio: ").append(String.format("%.1f%%", procesoManager.calcularEficienciaPromedio())).append("\n\n");
195  
196          // Procesos por tipo
197          reporte.append("DISTRIBUCIÓN POR TIPO\n");
198          reporte.append("-".repeat(25)).append("\n");
199          procesoManager.obtenerEstadisticasPorTipo().forEach((tipo, cantidad) ->
200                  reporte.append(tipo.getNombre()).append(": ").append(cantidad).append(" procesos\n")
201          );
202  
203          // Procesos con indicadores críticos
204          List<Proceso> procesosConProblemas = procesoManager.obtenerProcesosConIndicadoresCriticos();
205          if (!procesosConProblemas.isEmpty()) {
206              reporte.append("\nPROCESOS CON INDICADORES CRÍTICOS\n");
207              reporte.append("-".repeat(35)).append("\n");
208              procesosConProblemas.forEach(p ->
209                      reporte.append("• ").append(p.getNombre()).append(" (").append(p.getResponsable()).append(")\n")
210              );
211          }
212  
213          return reporte.toString();
214      }

284      private String evaluarRequisito(String requisito) {
285          // Evaluación simplificada basada en el estado del sistema
286          double cumplimiento = indicadorManager.calcularPorcentajeCumplimientoGeneral();
287  
288          if (cumplimiento >= 90) return "CONFORME ✓";
289          else if (cumplimiento >= 75) return "PARCIALMENTE CONFORME ⚠";
290          else return "NO CONFORME ✗";

336      private void exportarReporte(ActionEvent e) {
337          if (ultimoReporte == null || ultimoReporte.trim().isEmpty()) {
338              JOptionPane.showMessageDialog(this, "No hay reporte para exportar",
339                      "Advertencia", JOptionPane.WARNING_MESSAGE);
340              return;
341          }
342  
343          JFileChooser fileChooser = new JFileChooser();
344          fileChooser.setDialogTitle("Exportar Reporte");
345          fileChooser.setSelectedFile(new java.io.File("reporte_iso9001_" + LocalDate.now() + ".txt"));
346  
347          int userSelection = fileChooser.showSaveDialog(this);
348  
349          if (userSelection == JFileChooser.APPROVE_OPTION) {
350              try {
351                  java.io.File archivoDestino = fileChooser.getSelectedFile();
352  
353                  try (FileWriter writer = new FileWriter(archivoDestino)) {
354                      writer.write(ultimoReporte);
355                  }
356  
357                  JOptionPane.showMessageDialog(this,
358                          "Reporte exportado exitosamente a:\n" + archivoDestino.getAbsolutePath(),
359                          "Exportación Exitosa",
360                          JOptionPane.INFORMATION_MESSAGE);
361  
362              } catch (IOException ex) {
363                  JOptionPane.showMessageDialog(this,
364                          "Error al exportar el reporte: " + ex.getMessage(),
365                          "Error de Exportación",
366                          JOptionPane.ERROR_MESSAGE);
367              }
368          }
369      }

371      private void mostrarVistaPrevia(ActionEvent e) {
372          if (ultimoReporte == null || ultimoReporte.trim().isEmpty()) {
373              JOptionPane.showMessageDialog(this, "No hay reporte para mostrar",
374                      "Advertencia", JOptionPane.WARNING_MESSAGE);
375              return;
376          }
377  
378          JDialog dialogoVista = new JDialog(this, "Vista Previa de Impresión", true);
379          dialogoVista.setSize(700, 800);
380          dialogoVista.setLocationRelativeTo(this);
381  
382          JTextArea areaVista = new JTextArea(ultimoReporte);
383          areaVista.setEditable(false);
384          areaVista.setFont(new Font("Courier New", Font.PLAIN, 10));
385          areaVista.setBackground(Color.WHITE);
386          areaVista.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
387  
388          JScrollPane scrollVista = new JScrollPane(areaVista);
389          dialogoVista.add(scrollVista, BorderLayout.CENTER);
390  
391          JPanel panelBotones = new JPanel(new FlowLayout());
392          JButton btnCerrarVista = new JButton("Cerrar Vista");
393          btnCerrarVista.addActionListener(ev -> dialogoVista.dispose());
394          panelBotones.add(btnCerrarVista);
395  
396          dialogoVista.add(panelBotones, BorderLayout.SOUTH);
397          dialogoVista.setVisible(true);
398      }
