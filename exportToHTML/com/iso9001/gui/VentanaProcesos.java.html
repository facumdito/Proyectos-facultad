173      private void cargarDatos() {
174          modeloTabla.setRowCount(0);
175          List<Proceso> procesos = procesoManager.obtenerTodosProcesos();
176  
177          for (Proceso proceso : procesos) {
178              List<Indicador> indicadores = indicadorManager.obtenerIndicadoresPorProceso(proceso.getId());
179  
180              Object[] fila = {
181                      proceso.getId(),
182                      proceso.getNombre(),
183                      proceso.getTipo().getNombre(),
184                      proceso.getResponsable(),
185                      proceso.isActivo() ? "Sí" : "No",
186                      indicadores.size()
187              };
188  
189              modeloTabla.addRow(fila);
190          }
191      }

193      private void filtrarProcesos() {
194          String busqueda = txtBuscar.getText().trim();
195          TipoProceso tipoSeleccionado = (TipoProceso) comboTipo.getSelectedItem();
196  
197          modeloTabla.setRowCount(0);
198          List<Proceso> procesos = procesoManager.obtenerTodosProcesos();
199  
200          for (Proceso proceso : procesos) {
201              boolean coincide = true;
202  
203              // Filtrar por búsqueda de texto
204              if (!busqueda.isEmpty()) {
205                  coincide = proceso.getNombre().toLowerCase().contains(busqueda.toLowerCase()) ||
206                          proceso.getResponsable().toLowerCase().contains(busqueda.toLowerCase());
207              }
208  
209              // Filtrar por tipo
210              if (tipoSeleccionado != null) {
211                  coincide = coincide && proceso.getTipo() == tipoSeleccionado;
212              }
213  
214              if (coincide) {
215                  List<Indicador> indicadores = indicadorManager.obtenerIndicadoresPorProceso(proceso.getId());
216  
217                  Object[] fila = {
218                          proceso.getId(),
219                          proceso.getNombre(),
220                          proceso.getTipo().getNombre(),
221                          proceso.getResponsable(),
222                          proceso.isActivo() ? "Sí" : "No",
223                          indicadores.size()
224                  };
225  
226                  modeloTabla.addRow(fila);
227              }
228          }
229      }

237      private void mostrarDetallesProceso() {
238          int filaSeleccionada = tablaProcesos.getSelectedRow();
239          if (filaSeleccionada == -1) return;
240  
241          String procesoId = (String) modeloTabla.getValueAt(filaSeleccionada, 0);
242          Proceso proceso = procesoManager.obtenerProceso(procesoId);
243  
244          if (proceso != null) {
245              StringBuilder detalles = new StringBuilder();
246              detalles.append("DETALLES DEL PROCESO\n");
247              detalles.append("=".repeat(30)).append("\n\n");
248  
249              detalles.append("ID: ").append(proceso.getId()).append("\n");
250              detalles.append("Nombre: ").append(proceso.getNombre()).append("\n");
251              detalles.append("Tipo: ").append(proceso.getTipo().getNombre()).append("\n");
252              detalles.append("Responsable: ").append(proceso.getResponsable()).append("\n");
253              detalles.append("Estado: ").append(proceso.isActivo() ? "Activo" : "Inactivo").append("\n");
254              detalles.append("Fecha Creación: ").append(proceso.getFechaCreacion()).append("\n");
255              detalles.append("Última Revisión: ").append(proceso.getFechaUltimaRevision()).append("\n\n");
256  
257              if (proceso.getObjetivo() != null && !proceso.getObjetivo().isEmpty()) {
258                  detalles.append("Objetivo:\n").append(proceso.getObjetivo()).append("\n\n");
259              }
260  
261              if (proceso.getAlcance() != null && !proceso.getAlcance().isEmpty()) {
262                  detalles.append("Alcance:\n").append(proceso.getAlcance()).append("\n\n");
263              }
264  
265              if (proceso.getDescripcion() != null && !proceso.getDescripcion().isEmpty()) {
266                  detalles.append("Descripción:\n").append(proceso.getDescripcion()).append("\n\n");
267              }
268  
269              // Mostrar indicadores asociados
270              List<Indicador> indicadores = indicadorManager.obtenerIndicadoresPorProceso(proceso.getId());
271              if (!indicadores.isEmpty()) {
272                  detalles.append("INDICADORES ASOCIADOS\n");
273                  detalles.append("-".repeat(20)).append("\n");
274                  for (Indicador ind : indicadores) {
275                      detalles.append("• ").append(ind.getNombre()).append("\n");
276                      detalles.append("  Valor: ").append(String.format("%.2f", ind.getValorActual()));
277                      detalles.append(" / ").append(String.format("%.2f", ind.getValorObjetivo()));
278                      detalles.append(" (").append(ind.getEstadoSemaforo()).append(")\n");
279                  }
280              }
281  
282              areaDetalles.setText(detalles.toString());
283              areaDetalles.setCaretPosition(0);
284          }
285      }

366      private boolean validarFormulario(JTextField txtId, JTextField txtNombre, JTextField txtResponsable) {
367          if (txtId.getText().trim().isEmpty()) {
368              JOptionPane.showMessageDialog(this, "El ID es obligatorio", "Error", JOptionPane.ERROR_MESSAGE);
369              return false;
370          }
371  
372          if (procesoManager.existeProceso(txtId.getText().trim())) {
373              JOptionPane.showMessageDialog(this, "Ya existe un proceso con ese ID", "Error", JOptionPane.ERROR_MESSAGE);
374              return false;
375          }
376  
377          if (txtNombre.getText().trim().isEmpty()) {
378              JOptionPane.showMessageDialog(this, "El nombre es obligatorio", "Error", JOptionPane.ERROR_MESSAGE);
379              return false;
380          }
381  
382          if (txtResponsable.getText().trim().isEmpty()) {
383              JOptionPane.showMessageDialog(this, "El responsable es obligatorio", "Error", JOptionPane.ERROR_MESSAGE);
384              return false;
385          }
386  
387          return true;
388      }

499      private void desactivarProcesoSeleccionado() {
500          int filaSeleccionada = tablaProcesos.getSelectedRow();
501          if (filaSeleccionada == -1) {
502              JOptionPane.showMessageDialog(this, "Seleccione un proceso para desactivar",
503                      "Advertencia", JOptionPane.WARNING_MESSAGE);
504              return;
505          }
506  
507          String procesoId = (String) modeloTabla.getValueAt(filaSeleccionada, 0);
508  
509          int confirmacion = JOptionPane.showConfirmDialog(this,
510                  "¿Está seguro de que desea desactivar este proceso?",
511                  "Confirmar Desactivación",
512                  JOptionPane.YES_NO_OPTION);
513  
514          if (confirmacion == JOptionPane.YES_OPTION) {
515              procesoManager.eliminarProceso(procesoId);
516              cargarDatos();
517              areaDetalles.setText("Seleccione un proceso para ver los detalles...");
518  
519              JOptionPane.showMessageDialog(this, "Proceso desactivado exitosamente",
520                      "Éxito", JOptionPane.INFORMATION_MESSAGE);
521          }
522      }
